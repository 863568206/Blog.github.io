# 计算两点之间的距离


	double CalcPointDistance(const double &startX, const double &startY, const double &endX, const double &endY)
	{
	    double dx = endX - startX;
	    double dy = endY - startY;
	
	    return sqrt(dx * dx + dy * dy);
	}
